def performance_metric2(y_true, y_predict):
    """"""
    num1 = y_total = y_tot = y_res= 0
    #y_ture
    for i in y_true:
        y_total += i
        num1 += 1
    y_true_average = y_total / num1
    
    #
    for j in y_true:
        y_tot += pow(j-y_true_average,2)
        
    #
    y_array = np.array(y_true) - np.array(y_predict)
    for k in y_array:
        y_res += pow(k,2)
    score = 1 - y_res/y_tot

    return score
